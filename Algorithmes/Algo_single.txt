			public float getImpact(int Xc, int Yc){
				float D = Math.sqrt((x-Xc)²+(y-Yc)²);
				float Dmax = this.getDMax();
				
				float P = 0.0;
				
				if(Dmax<D){
					P = this.getPMax() * ((Dmax-D)/Dmax);
				}
				
				return P;		
			}
